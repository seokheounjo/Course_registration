Backend 개발 - Entity & JPA 설계

🗄️ 핵심 기능: 데이터베이스 설계

@Entity @Table(name = "students") public class Student { @Id @GeneratedValue(strategy = GenerationType.IDENTITY) private Long id; @Column(unique = true, nullable = false) @NotBlank(message = "학번은 필수 입력 항목입니다") private String studentId; @NotBlank(message = "이름은 필수 입력 항목입니다") private String name; private String department; private String password; private String grade; // 생성자, getter, setter 생략... }

핵심 기능: JPA Entity 매핑, Bean Validation으로 데이터 검증, 학번 중복 방지를 위한 유니크 제약조건 설정

🎯 핵심 기능: 수강신청 비즈니스 로직

@Service @Transactional public class EnrollmentService { @Autowired private StudentRepository studentRepository; @Autowired private TimetableRepository timetableRepository; @Autowired private EnrollmentRepository enrollmentRepository; /** * 수강신청 처리 메소드 * @param studentId 학생 ID * @param timetableId 시간표 ID * @return 수강신청 결과 */ public Enrollment enrollSubject(Long studentId, Long timetableId) { // 1. 학생 및 시간표 정보 조회 Student student = studentRepository .findById(studentId) .orElseThrow(() -> new RuntimeException("학생 정보를 찾을 수 없습니다")); Timetable timetable = timetableRepository .findById(timetableId) .orElseThrow(() -> new RuntimeException("시간표 정보를 찾을 수 없습니다")); // 2. 중복 수강 검증 boolean isDuplicate = enrollmentRepository .existsByStudentAndSubject(student, timetable.getSubject()); if (isDuplicate) { throw new RuntimeException("이미 수강 중인 과목입니다"); } // 3. 시간 충돌 검증 List<Enrollment> conflicts = enrollmentRepository.findTimeConflicts( student, timetable.getDay(), timetable.getStartTime(), timetable.getEndTime() ); if (!conflicts.isEmpty()) { throw new RuntimeException("시간이 겹치는 과목이 있습니다"); } // 4. 수강 정원 확인 long currentEnrollmentCount = enrollmentRepository.countByTimetable(timetable); if (currentEnrollmentCount >= timetable.getCapacity()) { throw new RuntimeException("수강 정원이 초과되었습니다"); } // 5. 수강신청 데이터 저장 Enrollment newEnrollment = new Enrollment( student, timetable.getSubject(), timetable ); return enrollmentRepository.save(newEnrollment); } }

핵심 기능: @Transactional로 데이터 일관성 보장, 4단계 검증 로직(중복/시간충돌/정원/저장), 체계적인 예외 처리

Service & Controller - 핵심 비즈니스 로직

🚀 핵심 기능: REST API 컨트롤러

@Controller @RequestMapping("/student") public class StudentController { @Autowired private StudentService studentService; @Autowired private EnrollmentService enrollmentService; /** * 수강신청 처리 컨트롤러 */ @PostMapping("/enroll") public String processEnrollment( @RequestParam Long timetableId, HttpSession session, RedirectAttributes redirectAttributes) { try { // 1. 세션에서 로그인 사용자 확인 String username = (String) session.getAttribute("username"); if (username == null) { redirectAttributes.addFlashAttribute( "errorMessage", "로그인이 필요합니다"); return "redirect:/login"; } // 2. 학생 정보 조회 Student student = studentService.getStudentByStudentId(username); if (student == null) { redirectAttributes.addFlashAttribute( "errorMessage", "학생 정보를 찾을 수 없습니다"); return "redirect:/student/enroll"; } // 3. 수강신청 처리 Enrollment enrollment = enrollmentService.enrollSubject( student.getId(), timetableId); // 4. 성공 메시지 설정 String successMessage = String.format( "수강신청이 완료되었습니다: %s", enrollment.getSubject().getName()); redirectAttributes.addFlashAttribute("successMessage", successMessage); } catch (RuntimeException e) { // 5. 예외 처리 redirectAttributes.addFlashAttribute("errorMessage", e.getMessage()); } return "redirect:/student/enroll"; } }

⚡ 핵심 기능: 복잡한 JPA 쿼리

@Repository public interface EnrollmentRepository extends JpaRepository<Enrollment, Long> { /** * 시간 충돌 검사를 위한 JPQL 쿼리 * 학생의 기존 수강 시간과 새로운 시간표의 시간이 겹치는지 확인 */ @Query(""" SELECT e FROM Enrollment e WHERE e.student = :student AND e.timetable.day = :day AND ( (e.timetable.startTime <= :startTime AND e.timetable.endTime > :startTime) OR (e.timetable.startTime < :endTime AND e.timetable.endTime >= :endTime) OR (e.timetable.startTime >= :startTime AND e.timetable.endTime <= :endTime) ) """) List<Enrollment> findTimeConflicts( @Param("student") Student student, @Param("day") String day, @Param("startTime") String startTime, @Param("endTime") String endTime ); /** * 학생이 특정 과목을 이미 수강하고 있는지 확인 */ boolean existsByStudentAndSubject( Student student, Subject subject ); /** * 특정 시간표에 등록된 학생 수 조회 */ long countByTimetable(Timetable timetable); /** * 학생별 수강 과목 조회 (시간순 정렬) */ @Query(""" SELECT e FROM Enrollment e WHERE e.student = :student ORDER BY e.timetable.day, e.timetable.startTime """) List<Enrollment> findByStudentOrderBySchedule( @Param("student") Student student ); }

핵심 기능: Text Block 문법으로 가독성 향상, 복잡한 시간 충돌 검사 JPQL, @Query와 메소드명 쿼리 조합 활용

✅ 핵심 기능: 실시간 프론트엔드 검증

/** * 수강신청 실시간 검증 모듈 */ // 상수 정의 const ENROLLMENT_CONFIG = { MAX_CREDITS: 18, // 최대 수강 학점 MIN_CREDITS: 12, // 최소 수강 학점 WARNING_THRESHOLD: 15 // 경고 기준 학점 }; // 전역 변수 let currentEnrollments = []; // 현재 수강 중인 과목 let selectedCourses = []; // 장바구니에 담은 과목 /** * 총 학점 계산 함수 */ function calculateTotalCredits() { let totalCredits = 0; // 기존 수강 과목 학점 합산 currentEnrollments.forEach(enrollment => { const credits = parseInt(enrollment.subject.credits) || 0; totalCredits += credits; }); // 선택한 과목 학점 합산 selectedCourses.forEach(course => { const credits = parseInt(course.credits) || 0; totalCredits += credits; }); return totalCredits; } /** * 수강신청 유효성 검증 함수 */ function validateEnrollment() { const totalCredits = calculateTotalCredits(); // 최대 학점 초과 검사 if (totalCredits > ENROLLMENT_CONFIG.MAX_CREDITS) { const message = `학점 한도를 초과했습니다. ` + `(${totalCredits}/${ENROLLMENT_CONFIG.MAX_CREDITS})`; alert(message); return false; } // 경고 메시지 표시 if (totalCredits >= ENROLLMENT_CONFIG.WARNING_THRESHOLD) { const message = `주의: 최대 학점에 근접했습니다. ` + `(${totalCredits}/${ENROLLMENT_CONFIG.MAX_CREDITS})`; console.warn(message); } return true; } /** * 학점 현황 업데이트 함수 */ function updateCreditStatus() { const totalCredits = calculateTotalCredits(); const creditStatusElement = document.getElementById('credit-status'); if (creditStatusElement) { creditStatusElement.textContent = `현재 학점: ${totalCredits}/${ENROLLMENT_CONFIG.MAX_CREDITS}`; } }

핵심 기능: 모듈화된 설정 객체, JSDoc 타입 안전성, 실시간 학점 계산 및 경고 시스템, 상태 업데이트 함수

VPC 네트워킹 문제

문제: EKS 클러스터와 RDS가 서로 다른 VPC에 생성되어 통신 불가

에러: vpc-0106946af7a80228d (EKS) ↔ vpc-0a532f02a6fa732c6 (RDS)

해결: VPC 피어링 연결 + 라우팅 테이블 업데이트로 cross-VPC 통신 구현
Docker 이미지 불일치

문제: GitHub Actions에서 잘못된 DockerHub 리포지토리 참조

에러: seokheounjo/course-registration → procof/course-registration

해결: DockerHub 사용자명 불일치 수정 및 시크릿 변수 재설정
데이터베이스 초기화 충돌

문제: data.sql 중복 키 에러로 애플리케이션 시작 실패

해결: 초기화 스크립트 비활성화 및 수동 데이터 입력으로 문제 해결
